apiVersion: apps/v1
kind: Deployment
metadata:
  labels:
    app: car-serv-deployment
  name: car-serv-deployment
  namespace: car-serv
spec:
  replicas: 2
  strategy:
    rollingUpdate:
      maxSurge: 1 #25% / so pod tăng
      maxUnavailable: 1 #25% / so pod giữ Unavailable, pod không sẵn sàng, pod sẽ được bỏ va thay thế bởi pod Surge
    type: RollingUpdate
  revisionHistoryLimit: 11
  selector:
    matchLabels:
      app: car-serv-deployment
  template:
    metadata:
      labels:
        app: car-serv-deployment
      namespace: car-serv
    spec:
      containers:
        - image: elroydevops/car-serv
          imagePullPolicy: Always
          name: car-serv
          ports:
            - containerPort: 80
              name: tcp
              protocol: TCP
