#Tạo replicated table production

✅ Nếu chỉ tạo ReplicatedMergeTree mà không có Distributed
Application phải tự biết:
insert node nào
=> production kém linh hoạt.

✅ Nếu chỉ có Distributed mà không có ReplicatedMergeTree
=> không có data backend.

✅ Kết luận production
✔ luôn tạo local trước
✔ rồi tạo distributed

CREATE TABLE logs ON CLUSTER prod_cluster
(
    ts DateTime,
    service String,
    message String
)
ENGINE = ReplicatedMergeTree(
    '/clickhouse/tables/{shard}/logs',
    '{replica}'
)
ORDER BY ts;

#Tạo distributed table
CREATE TABLE logs_all ON CLUSTER prod_cluster
AS logs
ENGINE = Distributed(prod_cluster, default, logs, rand());

# Logical backup
# cron backup
# shell script
#!/bin/bash

DATE=$(date +%F_%H-%M)

clickhouse-client --query "
BACKUP DATABASE default
TO Disk('minio_backup', 'daily_$DATE')
"

sudo chmod +x /opt/clickhouse-backup.sh

# crontab -e
0 2 * * * /opt/clickhouse-backup.sh

# restore
RESTORE DATABASE default AS restore_test
FROM Disk('minio_backup', 'daily_backup_001');

# cấu hình disk S3 trong ClickHouse
sudo vi /etc/clickhouse-server/config.d/backup_s3.xml
#metadata_path bắt buộc local.
<clickhouse>
  <storage_configuration>
    <disks>
      <minio_backup>
        <type>s3</type>
        <endpoint>http://192.168.181.100:9000/clickhouse-backup/</endpoint>
        <access_key_id>clickhousebackup</access_key_id>
        <secret_access_key>StrongPassword123</secret_access_key>
        <metadata_path>/data/clickhouse/disks/minio_backup/</metadata_path>
      </minio_backup>
    </disks>
  </storage_configuration>
</clickhouse>

#tao metadata dir
sudo mkdir -p /data/clickhouse/disks/minio_backup
sudo chown -R clickhouse:clickhouse /data/clickhouse/disks

sudo systemctl restart clickhouse-server

# kiểm tra disk nhận chưa
SELECT * FROM system.disks; # mong muon hien ra minio_backup

# backup thử 1 database
BACKUP DATABASE default
TO Disk('minio_backup', 'daily_backup_001');
