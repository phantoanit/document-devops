TRIỂN KHAI YUGABYTEDB ON-PREMISE

Tài liệu này hướng dẫn cài đặt trên 3 máy chủ VM.

# dat host name lan luot cho 3 vm yb1, yb2, yb3
sudo hostnamectl set-hostname yb1

#Tuning OS
vi /etc/sysctl.d/99-yugabyte.conf
vm.swappiness = 0
vm.dirty_background_ratio = 5
vm.dirty_ratio = 10
kernel.sched_migration_cost_ns = 5000000
kernel.numa_balancing = 0
net.core.somaxconn = 65535
net.ipv4.tcp_tw_reuse = 1

#Disable THP
sudo echo never > /sys/kernel/mm/transparent_hugepage/enabled
sudo echo never > /sys/kernel/mm/transparent_hugepage/defrag
#tạo service:
vi /etc/systemd/system/disable-thp.service
[Unit]
Description=Disable THP

[Service]
Type=oneshot
ExecStart=/bin/sh -c 'echo never > /sys/kernel/mm/transparent_hugepage/enabled'
ExecStart=/bin/sh -c 'echo never > /sys/kernel/mm/transparent_hugepage/defrag'

[Install]
WantedBy=multi-user.target

systemctl daemon-reload
systemctl enable disable-thp
systemctl start disable-thp

# tao truoc cac thu muc an thiet
mkdir -p /u01/colombo/yugabyte/data/{master,tserver}
mkdir -p /u01/colombo/var/logs/yugabyte
mkdir -p /u01/colombo/backups/yugabyte

# Đẩy bộ cài đặt vào trong thư mục /u01/colombo và giải nén
tar -xvzf yugabyte.tar.gz

#Sửa thông tin cấu hình trong file cấu hình da chuan bi san master.conf
master_addresses
replication_factor = 3
rpc_bind_addresses
# trong file cấu hình tserver.conf
tserver_master_addrs
rpc_bind_addresses
pgsql_proxy_bind_address

# Start Master
scripts/yb-master-start.sh
# Start Tserver
scripts/yb-tserver-start.sh 

# Kiểm tra leader master
/u01/colombo/yugabyte/bin/yb-admin -master_addresses 10.60.1.33:7100,10.60.3.33:7100,10.60.0.32:7100 list_all_masters

# Sau khi 3 master + 3 tserver lên, bạn nên chạy lệnh này 1 lần duy nhất để cố định topology:
yb-admin -master_addresses 10.60.1.33:7100,10.60.3.33:7100,10.60.0.32:7100 modify_placement_info cloud1.region1.zone1,cloud1.region1.zone2,cloud1.region1.zone3 3

# Check trạng thái của cụm
/u01/colombo/scripts/yb-status.sh
